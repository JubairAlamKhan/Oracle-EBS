SELECT 
    OPERATING_UNIT ,
    LEGAL_ENTITY,
    ORGANIZATION_ID,
    ORGANIZATION_CODE,
    ORGANIZATION_NAME,
    COUNT(*) 
        OVER(PARTITION BY OPERATING_UNIT ORDER BY OPERATING_UNIT 
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) ORG_SEQ
FROM ORG_ORGANIZATION_DEFINITIONS
ORDER BY OPERATING_UNIT,ORGANIZATION_ID;
