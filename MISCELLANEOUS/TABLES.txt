
---(1) ALL TABLES WITH DESCRIPTONS

SELECT 
    FA.APPLICATION_SHORT_NAME, 
    FT.TABLE_NAME, 
    FT.DESCRIPTION
FROM 
    FND_TABLES FT, 
    FND_APPLICATION FA
WHERE 
    FT.APPLICATION_ID = FA.APPLICATION_ID
    AND (APPLICATION_SHORT_NAME LIKE :APP_NAME
    OR TABLE_NAME LIKE '%\_'|| :APP_NAME ||'\_%' ESCAPE '\')
ORDER BY 1, 2;


SELECT
    DT.OWNER,
    DT.TABLE_NAME,
    FT.DESCRIPTION
FROM 
    DBA_ALL_TABLES DT,
    FND_TABLES FT
WHERE 
    DT.OWNER IN :APP_NAME
AND FT.TABLE_NAME(+) = DT.TABLE_NAME
ORDER BY 1,2,3 DESC;


---(2) ALL TABLES WITH NUMBER OF ROWS

SELECT 
    OWNER,
    TABLE_NAME,
    nvl(NUM_ROWS,0) "Number of Rows"
FROM 
    DBA_ALL_TABLES
WHERE 
    OWNER LIKE :APP_NAME
OR  TABLE_NAME LIKE '%'|| :APP_NAME ||'%'    
ORDER BY NUM_ROWS DESC NULLS LAST;

---(3) ALL POPULATED TABLES WITH DESCRIPTIONS AND NUMBER OF ROWS

SELECT
    DT.OWNER,
    DT.TABLE_NAME,
    FT.DESCRIPTION,
    DT.NUM_ROWS
FROM 
    DBA_ALL_TABLES DT,
    FND_TABLES FT
WHERE 
    (DT.OWNER IN :APP_NAME )
AND NVL(DT.NUM_ROWS,0) >1
AND FT.TABLE_NAME(+) = DT.TABLE_NAME
ORDER BY 4 DESC,2 ;
