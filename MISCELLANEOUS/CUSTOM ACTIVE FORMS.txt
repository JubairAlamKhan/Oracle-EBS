

---1 Getting all Oracle EBS Form name and corresponding fmb file name

SELECT 
    FFF.FUNCTION_NAME,
    FF.FORM_NAME,
    FFFT.USER_FUNCTION_NAME,
    FFF.FUNCTION_ID,
    FF.FORM_ID,
    FFF.APPLICATION_ID
FROM   
    FND_FORM FF,
    FND_FORM_FUNCTIONS FFF,
    FND_FORM_FUNCTIONS_TL FFFT
WHERE  
    FF.FORM_ID = FFF.FORM_ID
AND FFF.FUNCTION_ID = FFFT.FUNCTION_ID
AND FFFT.LANGUAGE = 'US'
AND FFF.APPLICATION_ID = 401   --Inventory Module
ORDER BY FFFT.USER_FUNCTION_NAME;
