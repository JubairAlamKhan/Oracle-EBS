
SELECT
   MSTC.STATUS_ID,
   MMST.STATUS_CODE STATUS,
   MSTC.TRANSACTION_TYPE_ID,
   MMT.DESCRIPTION TRANSACTION_TYPE,
   MSTC.IS_ALLOWED
FROM
   MTL_STATUS_TRANSACTION_CONTROL MSTC,
   MTL_MATERIAL_STATUSES_TL MMST,
   MTL_TRANSACTION_TYPES MMT
WHERE 
   MSTC.STATUS_ID = MMST.STATUS_ID
   AND MSTC.TRANSACTION_TYPE_ID = MMT.TRANSACTION_TYPE_ID  
ORDER BY 2,1;
