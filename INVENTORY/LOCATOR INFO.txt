SELECT 
    MIL.SEGMENT1 "Locator Segment1", 
    MIL.SEGMENT2 "Locator Segment2",
    MIL.SEGMENT3 "Locator Segment3", 
    MIL.SEGMENT4 "Locator Segment4", 
    MIL.SEGMENT5 "Locator Segment5",
    MIL.DESCRIPTION "Locator Description",
    MIL.INVENTORY_ITEM_ID "Item ID",
    MSI.DESCRIPTION "Item Description",
    OOD.ORGANIZATION_NAME "Org Name",
    MIL.SUBINVENTORY_CODE "Sub Inventory"
FROM 
    MTL_ITEM_LOCATIONS MIL,
    ORG_ORGANIZATION_DEFINITIONS OOD,
    (SELECT DISTINCT INVENTORY_ITEM_ID , DESCRIPTION
    FROM MTL_SYSTEM_ITEMS ) MSI
WHERE
    MIL.ORGANIZATION_ID = OOD.ORGANIZATION_ID
AND MIL.INVENTORY_ITEM_ID = MSI.INVENTORY_ITEM_ID(+)    
ORDER BY
    MIL.SUBINVENTORY_CODE,
    OOD.ORGANIZATION_NAME;
