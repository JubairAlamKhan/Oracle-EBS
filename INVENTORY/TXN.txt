
---(i) Transaction source type and transaction type

SELECT 
    TRANSACTION_SOURCE_TYPE_NAME,
    TRANSACTION_TYPE_NAME,
    DESCRIPTION,
    TRANSACTION_SOURCE_TYPE_ID,
    TRANSACTION_TYPE_ID
FROM 
    MTL_TRX_TYPES_VIEW
ORDER BY 1,2;

---(ii) Transaction reasons

SELECT
    REASON_ID,
    REASON_NAME,
    DESCRIPTION
FROM     
    MTL_TRANSACTION_REASONS_ALL_V    
ORDER BY 2;

---(iii) Serial number transactions

SELECT 
    TRANSACTION_ID,
    ORGANIZATION_ID,
    SUBINVENTORY_CODE,
    LOCATOR_ID,
    SERIAL_NUMBER,
    INVENTORY_ITEM_ID,
    TRANSACTION_DATE,
    TRANSACTION_SOURCE_ID,
    TRANSACTION_SOURCE_TYPE_ID,
    RECEIPT_ISSUE_TYPE,
    CUSTOMER_ID,
    PRODUCT_CODE,
    PRODUCT_TRANSACTION_ID
FROM
    MTL_UNIT_TRANSACTIONS
ORDER BY 2,3,4,5;

---(iv) All materials lot number transactions

SELECT 
    ORGANIZATION_ID,
    SUBINVENTORY_CODE,
    LOCATOR_ID,
    SERIAL_NUMBER,
    INVENTORY_ITEM_ID,
    TRANSACTION_ID,
    TRANSACTION_DATE,
    TRANSACTION_SOURCE_ID,
    TRANSACTION_SOURCE_TYPE_ID,
    RECEIPT_ISSUE_TYPE,
    CUSTOMER_ID,
    PRODUCT_CODE,
    PRODUCT_TRANSACTION_ID
FROM
    MTL_UNIT_TRANSACTIONS
ORDER BY 
    ORGANIZATION_ID,
    SUBINVENTORY_CODE,
    LOCATOR_ID,
    SERIAL_NUMBER,
    TRANSACTION_DATE;


---(v) Intercompany transaction flows of type shipping and procuring flows

SELECT 
   FROM_ORG_ID,
   TO_ORG_ID,
   START_ORG_ID,
   END_ORG_ID,
   HEADER_ID,
   LINE_NUMBER,
   FLOW_TYPE,
   ORGANIZATION_ID,
   QUALIFIER_CODE,
   QUALIFIER_VALUE_ID,
   START_DATE,
   END_DATE,
   ASSET_ITEM_PRICING_OPTION,
   EXPENSE_ITEM_PRICING_OPTION,
   FROM_ORGANIZATION_ID,
   TO_ORGANIZATION_ID,
   CUSTOMER_ID,
   ADDRESS_ID,
   CUSTOMER_SITE_ID,
   CUST_TRX_TYPE_ID,
   VENDOR_ID,
   VENDOR_SITE_ID,
   FREIGHT_CODE_COMBINATION_ID,
   INVENTORY_ACCRUAL_ACCOUNT_ID,
   EXPENSE_ACCRUAL_ACCOUNT_ID,
   INTERCOMPANY_COGS_ACCOUNT_ID,
   NEW_ACCOUNTING_FLAG,
   INV_CURRENCY_CODE
FROM    
    MTL_TRANSACTION_FLOWS_V 
ORDER BY    
   FROM_ORG_ID,
   TO_ORG_ID,
   HEADER_ID,
   LINE_NUMBER;

   TRANSACTION_ID,
   TRANSACTION_SOURCE_NAME,
   TRANSACTION_SOURCE,
   TRADING_PARTNER,
   PROJECT,

   TRANSACTION_QUANTITY,
   TRANSACTION_UOM,
   PRIMARY_QUANTITY,
   OBJECT_ID,
   OBJECT_TYPE,
   CREATED_BY,
   SOURCE_CODE,
   SOURCE_LINE_ID,
   TRANSACTION_TYPE_ID,
   LPN_NUMBER,
   TRANSFER_LPN_NUMBER,
   CONTENT_LPN_NUMBER,
   SUPPLIER_LOT_NUMBER,
   TRANSACTION_LOT_NUMBER

    MTL_MFG_LOTS_TRANS_V

   ORGANIZATION_ID,
