
---i Receive or issue quantity

SELECT 
    INVENTORY_ITEM_ID,
    SUM(TRANSACTION_QUANTITY)  TOTAL_RCV_QTY,
    TRANSACTION_UOM
FROM 
    MTL_MATERIAL_TRANSACTIONS
WHERE 
    (INVENTORY_ITEM_ID = :ITEM_ID OR :ITEM_ID IS NULL)
AND (ORGANIZATION_ID=:ORGANIZATION_ID OR :ORGANIZATION_ID IS NULL)
AND TRANSACTION_QUANTITY>0
GROUP BY INVENTORY_ITEM_ID, TRANSACTION_UOM;

---ii Order quantity

SELECT  
    SCHEDULE_SHIP_DATE, 
    ORDERED_ITEM, 
    SUM(ORDERED_QUANTITY) , 
    ORDER_QUANTITY_UOM
FROM    
    OE_ORDER_LINES_ALL
WHERE
    1=1
AND (INVENTORY_ITEM_ID=:YOUR_ITEM OR :YOUR_ITEM IS NULL)
AND (SHIP_FROM_ORG_ID=:ORGANIZATION_ID OR :ORGANIZATION_ID IS NULL)
AND (TO_CHAR(SCHEDULE_SHIP_DATE, 'DD-MON-RR') = :SHIPMENT_DATE OR :SHIPMENT_DATE IS NULL)
GROUP BY SCHEDULE_SHIP_DATE,ORDERED_ITEM, ORDER_QUANTITY_UOM;
