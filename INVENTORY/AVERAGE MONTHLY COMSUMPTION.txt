SELECT 
        SUM(TRANSACTION_QUANTITY)/30 
FROM 
        MTL_MATERIAL_TRANSACTIONS
WHERE 
        (INVENTORY_ITEM_ID=:YOUR_ITEM OR :YOUR_ITEM IS NULL)
AND     (ORGANIZATION_ID=:ORGANIZATION_ID OR :ORGANIZATION_ID IS NULL)
AND     TRANSACTION_QUANTITY<0
AND     TO_CHAR(TRANSACTION_DATE,'MON-RR') = :TXN_DATE
;
